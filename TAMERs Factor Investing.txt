    function animateValue(obj, start, end, duration, formatStr = "", isColor = false) {
      if(!obj) return;
      let startTimestamp = null;
      const step = (timestamp) => {
        if (!startTimestamp) startTimestamp = timestamp;
        const progress = Math.min((timestamp - startTimestamp) / duration, 1);
        // Easing out cubic
        const ease = 1 - Math.pow(1 - progress, 3);
        const current = start + ease * (end - start);
        
        let output = fmt(current);
        if(formatStr === "%") output += '<span style="font-size:18px">%</span>';
        else output += formatStr;
        
        obj.innerHTML = output;
        
        if(isColor) {
           obj.className = `num ${current < 0 ? 'danger' : ''}`;
        }
        
        if (progress < 1) window.requestAnimationFrame(step);
        else {
           obj.innerHTML = fmt(end) + (formatStr === "%" ? '<span style="font-size:18px">%</span>' : formatStr);
           if(isColor) obj.className = `num ${end < 0 ? 'danger' : ''}`;
        }
      };
      window.requestAnimationFrame(step);
    }

          <td>${fmt(nomMv,0)}</td>
          <td>${fmt(a.leverage)}x</td>
          <td>${a.category==='equity' ? fmt(a.betas.MKT) : '-'}</td>
          <td>${a.category==='equity' ? fmt(a.betas.SMB) : '-'}</td>
          <td>${a.category==='equity' ? fmt(a.betas.HML) : '-'}</td>
          <td>
            <button class="small btn-ghost" data-edit="${a.id}">載入</button>

      el('netValueDisplay').textContent = `淨值：$${fmt(totalNav, 0)}`;

    // ====== Core Calculation (Academic Engine) ======
    // Prevent global variable caching issues for animation
    let lastExpReturn = 0;
    let lastDrag = 0;
    let lastLev = 1;

    function calculate(){
      const btn = el('calcBtn');
      btn.classList.remove('glow');
      void btn.offsetWidth; // trigger reflow
      btn.classList.add('glow');

      const NAV = assets.reduce((s,a) => s + (a.qty||0)*(a.price||0), 0);
      if(NAV <= 0) {
        el('diagnostics').innerHTML = '<span style="color:var(--danger)">錯誤：帳戶淨值為零，請輸入持倉價格與數量。</span>';
        return;
      }

      // 1. Get Params
      const Rf = getVal('p_cash') / 100;
      const sigmaMarket = getVal('p_vol') / 100;
      const friction = getVal('p_fric') / 100;
      
      const premiums = {
        MKT: getVal('p_MKT') / 100, SMB: getVal('p_SMB') / 100,
        HML: getVal('p_HML') / 100, RMW: getVal('p_RMW') / 100, CMA: getVal('p_CMA') / 100
      };

      // 2. Aggregate Exposures & Leverage
      let totalNominal = 0;
      let totalEquityNominal = 0;
      let expPort = { MKT:0, SMB:0, HML:0, RMW:0, CMA:0 };
      
      let uniqueEquities = new Set();
      let maxConcentrationNominal = 0;

      assets.forEach(a => {
        const mv = (a.qty || 0) * (a.price || 0);
        const nominal = mv * (a.leverage || 1);
        totalNominal += nominal;
        
        if(nominal > maxConcentrationNominal) maxConcentrationNominal = nominal;

        if(a.category === 'equity') {
          totalEquityNominal += nominal;
          uniqueEquities.add(a.symbol);
          factorList.forEach(f => {
            // Port Beta = Sum( Asset_Weight_by_NAV * Asset_Beta )
            expPort[f] += (nominal / NAV) * (a.betas[f] || 0);
          });
        }
      });

      const effectiveLeverage = totalNominal / NAV;

      // 3. Expected Arithmetic Return calculations
      let factorExpectedReturn = 0;
      factorList.forEach(f => { factorExpectedReturn += expPort[f] * premiums[f]; });

      const borrowingCost = Math.max(0, effectiveLeverage - 1) * Rf;
      const arithmeticReturn = Rf + factorExpectedReturn - borrowingCost - friction;

      // 4. Volatility Drag (Approximation for Leveraged Geometric Return)
      // Drag = 0.5 * (Leverage * Sigma)^2
      const portVol = (totalEquityNominal / NAV) * sigmaMarket; // Approximation: scaling market vol by equity leverage
      const volDrag = 0.5 * Math.pow(portVol, 2);
      
      const geometricReturn = arithmeticReturn - volDrag;

      // 5. Safety Scoring Engine
      const concPct = maxConcentrationNominal / totalNominal; // Max position size
      let scoreConc = Math.max(0, 100 - (concPct * 100 * 1.5)); // Penalize if > 33%
      
      let scoreLev = Math.max(0, 100 - ((effectiveLeverage - 1) * 40)); // Penalize high leverage
      
      let penaltyDiversification = 0;
      if(uniqueEquities.size < 5 && uniqueEquities.size > 0) {
        penaltyDiversification = (5 - uniqueEquities.size) * 10; // -10 points per missing asset under 5
      }

      let safety = Math.round(scoreConc * 0.4 + scoreLev * 0.6) - penaltyDiversification;
      safety = Math.max(0, Math.min(100, safety));

      // 6. UI Updates (Animations)
      const targetRet = geometricReturn * 100;
      const targetDrag = volDrag * 100;
      
      animateValue(el('expReturn'), lastExpReturn, targetRet, 1000, "%", true);
      animateValue(el('dragLoss'), lastDrag, targetDrag, 1000, "%");
      animateValue(el('totalLev'), lastLev, effectiveLeverage, 1000, "x");
      
      lastExpReturn = targetRet; lastDrag = targetDrag; lastLev = effectiveLeverage;

      // Update Radars & Texts
      updateRadar([expPort.MKT, expPort.SMB, expPort.HML, expPort.RMW, expPort.CMA]);
      factorList.forEach(f => { el(`txt_${f}`).textContent = fmt(expPort[f]); });

      // Safety UI
      const sBox = el('safetyScore');
      sBox.textContent = `系統安全：${safety}`;
      sBox.className = 'pill ' + (safety > 75 ? 'score-high' : safety > 45 ? 'score-mid' : 'score-low');

      // 7. Academic Diagnostics Generation
      let diag = `<div class="diag-item"><strong>投資組合結構：</strong> 總名義曝險 $${fmt(totalNominal,0)} (實質槓桿 ${fmt(effectiveLeverage)}x)。</div>`;
      
      if(effectiveLeverage > 1.5) {
        diag += `<div class="diag-item" style="color:var(--warning)"><strong>⚠️ 槓桿警告：</strong> 實質槓桿達 ${fmt(effectiveLeverage)}x，預估造成每年 ${fmt(targetDrag)}% 的波動率拖累 (Volatility Drag)。在劇烈震盪市場中，複利損耗將顯著侵蝕淨值。</div>`;
      }
      if(uniqueEquities.size > 0 && uniqueEquities.size < 5) {
        diag += `<div class="diag-item" style="color:var(--warning)"><strong>⚠️ 非系統風險：</strong> 股票資產僅 ${uniqueEquities.size} 檔。Fama-French 模型基於高度分散的前提，集中持股將導致「個股風險 (Idiosyncratic Risk)」主導回報，因子溢酬可能失效。</div>`;
      }
      if(concPct > 0.4) {
        diag += `<div class="diag-item" style="color:var(--danger)"><strong>🚨 集中度過高：</strong> 最大單一資產佔比達 ${fmt(concPct*100)}%，面臨嚴重尾部風險。</div>`;
      }
      if(expPort.MKT < 0.5 && totalEquityNominal > 0) {
         diag += `<div class="diag-item"><strong>ℹ️ 低市場曝險：</strong> 組合 MKT Beta 偏低，可能錯失大盤基準報酬，請確認是否為刻意的對沖配置。</div>`;
      }
      
      if(diag.indexOf('⚠️') === -1 && diag.indexOf('🚨') === -1) {
        diag += `<div class="diag-item" style="color:var(--success)"><strong>✅ 狀態健康：</strong> 槓桿控制得當且風險分散，幾何預期報酬具備學術統計上的合理性。</div>`;
      }

      el('diagnostics').innerHTML = diag;
      save();
    }
